```c++
//代码
#include<iostream>
//打印一行的函数
void Print(int number,char c){
    while(number--){
        std::cout<<c<<endl;
    }
}
int main(){
    int number;
    char c;
    int hnumber;
    //数字和字符之间间隔一个空格的操作不知道
    cin>>number>>c;
    std::cout<<Print(number,c)
    if(number%2) hnumber = number/2 + 1;
    else hnumber = number/2;
    while(hnumber--)
        Print(2,c);
    std::cout<<Print(number,c);
    return 0;
}
//结尾无空行没有处理
```
