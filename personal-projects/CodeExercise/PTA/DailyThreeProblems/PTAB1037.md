```c++
//代码简写
//如果使用一个数组实现循环还是挺简单的
#include<iostream>
#include<cmath>
using namespace std;

int Galleon[2],Sickle[2],Knut[2],money[2];
int size;
int i = 0;
size = pow(10,20);
char all[size];

while((all[i] = getchar() != '/n')) i++;
for(int y = 0;y <= i;y++){
    if(all[y] != '.') Galleon[0] += (all[y]-'0');
    else break;
}
//这个y+2的边界条件需要写个程序试试看
for(int z = y+2;z <= i;z++){
    if(all[z] != '.') Sickle[0] += (all[z]-'0');
    else break;
}

for(int m = z+2;m <= i;m++){
    if(all[m] != '.') Knut[0] += (all[m]-'0');
    else break;
}

for(int n = m+3;n <= i;n++){
    if(all[n] != '.') Galleon[1] += (all[n]-'0');
    else break;
}

for(int o = n+2;o <= i;o++){
    if(all[o] != '.') Sickle[1] += (all[0]-'0');
    else break;
}

for(int p = o+2;p <= i;p++){
    if(all[p] != '.') Knut[1] += (all[p]-'0');
    else break;
}
//输出的值一定是绝对值
//先求绝对值
//然后比较各项输出符号,避免对负数直接操作
money[0] = CalValue(Galleon[0],Sickle[0],Knut[0]);
money[1] = CalValue(Galleon[1],Sickle[1],Knut[1]);

PrintValue(money[1]-money[0]);//这里应该不需要考虑正负数输出的情况,输出应该是根据具体情况输出的

int CalValue(int Galleon,int Sickle,int Knut){
    int value;
    value = Galleon*17*29 + Sickle*29 + Knut;
}
void PrintValue(int value){
    Galleon = value/(17*29);
    Sickle = (value-Galleon*17*29)/29;
    Knut = value - (value-Galleon*17*29) - Sickle*29;
    cout<<Galleon<<'.'<<Sickle<<':'<<Knut;
}
```
